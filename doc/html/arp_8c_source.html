64 int
65 main (int argc, char **argv)
66 {
67  int i, status, frame_length, sd, bytes;
68  char *interface, *target, *src_ip;
69  arp_hdr arphdr;
70  uint8_t *src_mac, *dst_mac, *ether_frame;
71  struct addrinfo hints, *res;
72  struct sockaddr_in *ipv4;
73  struct sockaddr_ll device;
74  struct ifreq ifr;
75 
76  // Allocate memory for various arrays.
77  src_mac = allocate_ustrmem (6);
78  dst_mac = allocate_ustrmem (6);
79  ether_frame = allocate_ustrmem (IP_MAXPACKET);
80  interface = allocate_strmem (40);
81  target = allocate_strmem (40);
82  src_ip = allocate_strmem (INET_ADDRSTRLEN);
83 
84  // Interface to send packet through.
85  strcpy (interface, "eth0");
86 
87  // Submit request for a socket descriptor to look up interface.
88  if ((sd = socket (AF_INET, SOCK_RAW, IPPROTO_RAW)) < 0) {
89  perror ("socket() failed to get socket descriptor for using ioctl() ");
90  exit (EXIT_FAILURE);
91  }
92 
93  // Use ioctl() to look up interface name and get its MAC address.
94  memset (&ifr, 0, sizeof (ifr));
95  snprintf (ifr.ifr_name, sizeof (ifr.ifr_name), "%s", interface);
96  if (ioctl (sd, SIOCGIFHWADDR, &ifr) < 0) {
97  perror ("ioctl() failed to get source MAC address ");
98  return (EXIT_FAILURE);
99  }
100  close (sd);
101 
102  // Copy source MAC address.
103  memcpy (src_mac, ifr.ifr_hwaddr.sa_data, 6 * sizeof (uint8_t));
104 
105  // Report source MAC address to stdout.
106  printf ("MAC address for interface %s is ", interface);
107  for (i=0; i<5; i++) {
108  printf ("%02x:", src_mac[i]);
109  }
110  printf ("%02x\n", src_mac[5]);
111 
112  // Find interface index from interface name and store index in
113  // struct sockaddr_ll device, which will be used as an argument of sendto().
114  memset (&device, 0, sizeof (device));
115  if ((device.sll_ifindex = if_nametoindex (interface)) == 0) {
116  perror ("if_nametoindex() failed to obtain interface index ");
117  exit (EXIT_FAILURE);
118  }
119  printf ("Index for interface %s is %i\n", interface, device.sll_ifindex);
120 
121  // Set destination MAC address: broadcast address
122  memset (dst_mac, 0xff, 6 * sizeof (uint8_t));
123 
124  // Source IPv4 address: you need to fill this out
125  strcpy (src_ip, "192.168.1.116");
126 
127  // Destination URL or IPv4 address (must be a link-local node): you need to fill this out
128  strcpy (target, "192.168.1.1");
129 
130  // Fill out hints for getaddrinfo().
131  memset (&hints, 0, sizeof (struct addrinfo));
132  hints.ai_family = AF_INET;
133  hints.ai_socktype = SOCK_STREAM;
134  hints.ai_flags = hints.ai_flags | AI_CANONNAME;
135 
136  // Source IP address
137  if ((status = inet_pton (AF_INET, src_ip, &arphdr.sender_ip)) != 1) {
138  fprintf (stderr, "inet_pton() failed for source IP address.\nError message: %s", strerror (status));
139  exit (EXIT_FAILURE);
140  }
141 
142  // Resolve target using getaddrinfo().
143  if ((status = getaddrinfo (target, NULL, &hints, &res)) != 0) {
144  fprintf (stderr, "getaddrinfo() failed: %s\n", gai_strerror (status));
145  exit (EXIT_FAILURE);
146  }
147  ipv4 = (struct sockaddr_in *) res->ai_addr;
148  memcpy (&arphdr.target_ip, &ipv4->sin_addr, 4 * sizeof (uint8_t));
149  freeaddrinfo (res);
150 
151  // Fill out sockaddr_ll.
152  device.sll_family = AF_PACKET;
153  memcpy (device.sll_addr, src_mac, 6 * sizeof (uint8_t));
154  device.sll_halen = 6;
155 
156  // ARP header
157 
158  // Hardware type (16 bits): 1 for ethernet
159  arphdr.htype = htons (1);
160 
161  // Protocol type (16 bits): 2048 for IP
162  arphdr.ptype = htons (ETH_P_IP);
163 
164  // Hardware address length (8 bits): 6 bytes for MAC address
165  arphdr.hlen = 6;
166 
167  // Protocol address length (8 bits): 4 bytes for IPv4 address
168  arphdr.plen = 4;
169 
170  // OpCode: 1 for ARP request
171  arphdr.opcode = htons (ARPOP_REQUEST);
172 
173  // Sender hardware address (48 bits): MAC address
174  memcpy (&arphdr.sender_mac, src_mac, 6 * sizeof (uint8_t));
175 
176  // Sender protocol address (32 bits)
177  // See getaddrinfo() resolution of src_ip.
178 
179  // Target hardware address (48 bits): zero, since we don't know it yet.
180  memset (&arphdr.target_mac, 0, 6 * sizeof (uint8_t));
181 
182  // Target protocol address (32 bits)
183  // See getaddrinfo() resolution of target.
184 
185  // Fill out ethernet frame header.
186 
187  // Ethernet frame length = ethernet header (MAC + MAC + ethernet type) + ethernet data (ARP header)
188  frame_length = 6 + 6 + 2 + ARP_HDRLEN;
189 
190  // Destination and Source MAC addresses
191  memcpy (ether_frame, dst_mac, 6 * sizeof (uint8_t));
192  memcpy (ether_frame + 6, src_mac, 6 * sizeof (uint8_t));
193 
194  // Next is ethernet type code (ETH_P_ARP for ARP).
195  // http://www.iana.org/assignments/ethernet-numbers
196  ether_frame[12] = ETH_P_ARP / 256;
197  ether_frame[13] = ETH_P_ARP % 256;
198 
199  // Next is ethernet frame data (ARP header).
200 
201  // ARP header
202  memcpy (ether_frame + ETH_HDRLEN, &arphdr, ARP_HDRLEN * sizeof (uint8_t));
203 
204  // Submit request for a raw socket descriptor.
205  if ((sd = socket (PF_PACKET, SOCK_RAW, htons (ETH_P_ALL))) < 0) {
206  perror ("socket() failed ");
207  exit (EXIT_FAILURE);
208  }
209 
210  // Send ethernet frame to socket.
211  if ((bytes = sendto (sd, ether_frame, frame_length, 0, (struct sockaddr *) &device, sizeof (device))) <= 0) {
212  perror ("sendto() failed");
213  exit (EXIT_FAILURE);
214  }
215 
216  // Close socket descriptor.
217  close (sd);
218 
219  // Free allocated memory.
220  free (src_mac);
221  free (dst_mac);
222  free (ether_frame);
223  free (interface);
224  free (target);
225  free (src_ip);
226 
227  return (EXIT_SUCCESS);
228 }
